Quick Start

# Download and extract
wget https://iscan.life/download/iscan_life_mvp_v0.1.zip
unzip iscan_life_mvp_v0.1.zip
cd iscan_life

# Install
./setup.sh

# Start using
iscan note       # Add a quick note
iscan dash       # View dashboard
iscan search     # Search your knowledge
